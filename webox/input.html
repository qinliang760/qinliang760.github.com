引用资源文件


<link rel="stylesheet" href="css/input/input.css" />
<script src="js/input/jquery.input.js"></script>

HTML

/*普通输入框*/ <input type="text" class="form_input" value=""/> /*提示文字输入框*/ <input type="text" class="tip_input" value="我是提示性的文字"/> /*失效输入框*/ <input type="text" class="disabled_input" value="此输入框无效"/>

JS

/*普通输入框*/ $(".form_input").input(); /*提示文字输入框*/ $(".tip_input").input({tipTxt:true}); /*失效输入框*/ $(".disabled_input").input({disabled:true});
