HTML

<input type="text" id="autoInput1"> <input type="text" id="autoInput2"> <input type="text" id="autoInput3">

JS

$(function(){ //本地数据 $("#autoInput1").autocomplete(["c++", "java", "秦亮", "coldfusion", "javascript", "asp", "ruby", "python", "c", "scala", "groovy", "haskell", "pearl"],{ minChars: 0,//双击弹出提示框 max: 4,//最大显示数 multiple: true, scroll: true, scrollHeight: 300 }); //ajax得到的数组数据 $("#autoInput2").autocomplete("php/autocomplete/search.php"); //ajax得到的json数据 $("#autoInput3").autocomplete('php/autocomplete/emails.php', { width:200, multiple: true, dataType: "json", parse: function(data) { return $.map(data, function(row) { return { data: row, value: row.name, result: row.name + " <" + row.to + ">" } }); }, formatItem: function(item) { return item.name + " &lt;" + item.to + "&gt"; } }).result(function(e, item) { $("body").append("<p>selected " + format(item) + "</p>"); }); })
