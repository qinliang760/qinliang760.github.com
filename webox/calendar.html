Calendar组件说明

这是一个基于jquery的日历组件,可以定制显示的月份数目，可以用来选择连续的时间段，控制可选择的时间段.
官网参考:http://www.eyecon.ro/datepicker/

JS


$(function(){
	$('#calendar1').DatePicker({
		flat: true,
		format: 'Y-m-d',
		date: [new Date(), new Date()],
		calendars: 2,
		mode: 'range',
		starts: 1,
		onChange: function(formated) {
			$('#field1 span').get(0).innerHTML = formated.join(' ~ ');
			$('#frm_date').attr('value',formated);
		}
	});
	var state = false;
	$('#field1').bind('click', function(){
		$('#calendar1').stop().animate({height: state ? 0 : $('#calendar1 div.datepicker').get(0).offsetHeight}, 500);
		state = !state;
		return false;
	});
	$('#calendar1 div.datepicker').css('position', 'absolute');  

	$('.datepicker_input').DatePicker({
		format:'Y-m-d',
		date: $('#inputDate').val(),
		current: $('#inputDate').val(),
		starts: 1,
		position: 'bottom',
		onBeforeShow: function(){
			$('#inputDate').DatePickerSetDate($('#inputDate').val(), true);
		},
		onChange: function(formated, dates){
			$('#inputDate').val(formated);
			$('#inputDate').DatePickerHide();
		}
	});	 
})
