JS

$(function(){ //galile中用的层级展现 $("#autoInput4").autocomplete('php/autocomplete/data.php', { width:200, multiple: false, dataType: "json", parse: function(data){ return $.map(data, function(row){ return { data: row, value: row.category, result: "" } });}, formatItem: function(item) { var itemStr='<dl class="autoDl"><dt class="category">'+item.category+'</dt>'; var itemDlEnd='</dl>'; for(i=0;i<item.childs.length;i++){ itemStr+='<dd class="subCategory"><a href="'+item.childs[i].href+'">'+item.childs[i].subCategory+'</a></dd>'; } return itemStr+itemDlEnd; } }) $(".autoDl > dd").live("mouseover",function(){ $(".ac_over2").removeClass("ac_over2"); $(this).addClass("ac_over2"); }) })
